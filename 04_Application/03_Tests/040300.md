En cours de déploiement...


<div id="decision-tree" style="font-family:Avenir; max-width:600px; margin:auto;">
  <h3 style="color:#006699;">Choisissez votre besoin</h3>
  <div id="question"></div>
  <div id="options" style="margin-top:10px;"></div>
</div>

<script>
const tree = {
  question: "Quel type de données souhaitez-vous gérer ?",
  options: [
    { text: "Données relationnelles", next: {
        question: "Volume attendu ?",
        options: [
          { text: "Petit à moyen", result: "Utilisez PostgreSQL" },
          { text: "Très grand", result: "Utilisez Oracle ou SQL Server" }
        ]
      }
    },
    { text: "Données non structurées", next: {
        question: "Fréquence d'accès ?",
        options: [
          { text: "Élevée", result: "Utilisez MongoDB" },
          { text: "Faible", result: "Utilisez CouchDB" }
        ]
      }
    }
  ]
};

let currentNode = tree;

function renderNode(node) {
  const qDiv = document.getElementById("question");
  const oDiv = document.getElementById("options");
  qDiv.innerHTML = `<p style="color:#006699;font-weight:bold;">${node.question || "Résultat"}</p>`;
  oDiv.innerHTML = "";

  if (node.options) {
    node.options.forEach(opt => {
      const btn = document.createElement("button");
      btn.textContent = opt.text;
      btn.style.cssText = "background:#006699;color:#fff;border:none;padding:8px;margin:5px;cursor:pointer;";
      btn.onclick = () => renderNode(opt.next || { question: opt.result });
      oDiv.appendChild(btn);
    });
  }
}

renderNode(currentNode);
</script>
